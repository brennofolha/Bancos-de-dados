if not exists - siginifica basicamente, se nao existir executa o codigo mas se existir  nao executa

if exists - é o contrario , se existir executa  mas se nao existir nao executa

unique - nao deixa ter duas coisas com o mesmo nome ou identico 

unsigned - nao deixa o campo ser negativo, so pode ser positivo ou zero

exemplo:
create table if not exists cursos(
	nome varchar(30) not null unique,
    	descricao text,
    	carga int unsigned,
    	total_aulas int unsigned,
    	ano year default '2023' 
) default charset = utf8;
